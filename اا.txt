    <script>
        // --- البيانات والحالة ---
        let curUser = null;
        let appData = {
            students: [],
            attendance: {},
            grades: {},
            settings: { hw: 10, part: 10, mid: 30, final: 50 }
        };

        // --- نظام التنبيهات ---
        function showToast(msg) {
            const t = document.getElementById('toast');
            t.innerText = msg;
            t.className = 'show';
            setTimeout(() => t.className = '', 3000);
        }

        // --- المصادقة ---
        function toggleAuth() {
            const l = document.getElementById('login-box');
            const s = document.getElementById('signup-box');
            l.style.display = l.style.display === 'none' ? 'block' : 'none';
            s.style.display = s.style.display === 'none' ? 'block' : 'none';
        }

        function signup() {
            const u = document.getElementById('regUser').value.trim();
            const p = document.getElementById('regPass').value;
            const n = document.getElementById('regName').value;
            const s = document.getElementById('regSub').value;
            if(!u || !p || !n || !s) return alert('أكمل جميع الحقول');

            let users = JSON.parse(localStorage.getItem('LMS_USERS') || '{}');
            if(users[u]) return alert('اسم المستخدم موجود مسبقاً');

            users[u] = { name: n, sub: s, pass: p };
            localStorage.setItem('LMS_USERS', JSON.stringify(users));
            localStorage.setItem(`LMS_DATA_${u}`, JSON.stringify(appData));
            showToast('تم إنشاء الحساب بنجاح');
            toggleAuth();
        }

        function login() {
            const u = document.getElementById('logUser').value.trim();
            const p = document.getElementById('logPass').value;
            let users = JSON.parse(localStorage.getItem('LMS_USERS') || '{}');

            if(users[u] && users[u].pass === p) {
                curUser = u;
                appData = JSON.parse(localStorage.getItem(`LMS_DATA_${u}`));
                document.getElementById('displayUserName').innerText = users[u].name;
                document.getElementById('displayUserSub').innerText = users[u].sub;
                document.getElementById('auth-screen').style.display = 'none';
                document.getElementById('app-screen').style.display = 'flex';
                updateDashboard();
                loadSettingsToUI();
            } else {
                alert('خطأ في بيانات الدخول');
            }
        }

        function logout() { location.reload(); }

        // --- التنقل ---
        function nav(sectionId) {
            document.querySelectorAll('.section').forEach(s => s.classList.remove('active'));
            document.querySelectorAll('.nav-item').forEach(i => i.classList.remove('active'));
            document.getElementById(sectionId).classList.add('active');
            event.currentTarget.classList.add('active');

            if(sectionId === 'students') renderStudents();
            if(sectionId === 'grades') renderGrades();
            if(sectionId === 'interaction') renderInteraction();
            if(sectionId === 'dashboard') updateDashboard();
        }

        // --- إدارة الطلاب ---
        function addStudent() {
            const name = document.getElementById('stdName').value;
            const id = document.getElementById('stdId').value;
            if(!name || !id) return showToast('أدخل بيانات الطالب');
            
            appData.students.push({ id, name, partScore: 0 });
            saveData();
            renderStudents();
            document.getElementById('stdName').value = '';
            document.getElementById('stdId').value = '';
        }

        function renderStudents() {
            const query = document.getElementById('searchStd').value.toLowerCase();
            const tbody = document.querySelector('#stdTable tbody');
            tbody.innerHTML = '';
            
            appData.students.filter(s => s.name.toLowerCase().includes(query) || s.id.includes(query))
            .forEach(s => {
                tbody.innerHTML += `<tr>
                    <td>${s.id}</td>
                    <td>${s.name}</td>
                    <td><button class="btn btn-danger" onclick="deleteStudent('${s.id}')"><i class="fas fa-trash"></i></button></td>
                </tr>`;
            });
        }

        function deleteStudent(id) {
            if(confirm('هل أنت متأكد من حذف الطالب وجميع بياناته؟')) {
                appData.students = appData.students.filter(s => s.id !== id);
                delete appData.grades[id];
                saveData();
                renderStudents();
            }
        }

        // --- سجل الدرجات ---
        function renderGrades() {
            const head = document.getElementById('gradesHead');
            const s = appData.settings;
            head.innerHTML = `<th>اسم الطالب</th><th>واجبات (${s.hw})</th><th>تفاعل (${s.part})</th><th>نصفي (${s.mid})</th><th>نهائي (${s.final})</th><th>المجموع</th><th>الحالة</th>`;

            const tbody = document.querySelector('#gradesTable tbody');
            tbody.innerHTML = '';

            appData.students.forEach(std => {
                const g = appData.grades[std.id] || { h:0, m:0, f:0 };
                const p = std.partScore || 0;
                const total = parseFloat(g.h||0) + parseFloat(p||0) + parseFloat(g.m||0) + parseFloat(g.f||0);
                
                tbody.innerHTML += `<tr>
                    <td style="text-align:right">${std.name}</td>
                    <td><input type="number" class="grade-input" oninput="liveCalc(this,'h','${std.id}')" value="${g.h}" max="${s.hw}"></td>
                    <td><input type="number" class="grade-input" oninput="liveCalc(this,'p','${std.id}')" value="${p}" max="${s.part}"></td>
                    <td><input type="number" class="grade-input" oninput="liveCalc(this,'m','${std.id}')" value="${g.m}" max="${s.mid}"></td>
                    <td><input type="number" class="grade-input" oninput="liveCalc(this,'f','${std.id}')" value="${g.f}" max="${s.final}"></td>
                    <td class="total-cell" style="font-weight:bold">${total}</td>
                    <td class="status-cell">${total >= 50 ? '<span style="color:green">ناجح</span>' : '<span style="color:red">راسب</span>'}</td>
                </tr>`;
            });
        }

        function liveCalc(input, type, stdId) {
            const max = appData.settings[type === 'h' ? 'hw' : type === 'p' ? 'part' : type === 'm' ? 'mid' : 'final'];
            if(parseFloat(input.value) > max) {
                alert(`الدرجة العظمى هي ${max}`);
                input.value = max;
            }

            const row = input.closest('tr');
            const inputs = row.querySelectorAll('.grade-input');
            let sum = 0;
            inputs.forEach(i => sum += parseFloat(i.value || 0));
            
            row.querySelector('.total-cell').innerText = sum;
            row.querySelector('.status-cell').innerHTML = sum >= 50 ? '<span style="color:green">ناجح</span>' : '<span style="color:red">راسب</span>';
        }

        function saveAllGrades() {
            const rows = document.querySelectorAll('#gradesTable tbody tr');
            rows.forEach((row, index) => {
                const std = appData.students[index];
                const inps = row.querySelectorAll('.grade-input');
                
                appData.grades[std.id] = {
                    h: inps[0].value,
                    m: inps[2].value,
                    f: inps[3].value
                };
                std.partScore = inps[1].value;
            });
            saveData();
            showToast('تم حفظ جميع الدرجات بنجاح');
        }

        // --- التفاعل ---
        function renderInteraction() {
            const container = document.getElementById('interCards');
            container.innerHTML = '';
            appData.students.forEach(s => {
                container.innerHTML += `
                <div class="card" style="text-align:center">
                    <h4>${s.name}</h4>
                    <h2 id="part-${s.id}" style="color:var(--accent)">${s.partScore}</h2>
                    <div style="display:flex; gap:10px; justify-content:center">
                        <button class="btn btn-success" onclick="updatePart('${s.id}', 1)">+1</button>
                        <button class="btn btn-danger" onclick="updatePart('${s.id}', -1)">-1</button>
                    </div>
                </div>`;
            });
        }

        function updatePart(id, val) {
            const std = appData.students.find(s => s.id === id);
            let newVal = parseFloat(std.partScore || 0) + val;
            if(newVal < 0) newVal = 0;
            if(newVal > appData.settings.part) return alert('وصل الطالب للحد الأقصى للتفاعل');
            
            std.partScore = newVal;
            document.getElementById(`part-${id}`).innerText = newVal;
            saveData();
        }

        // --- الحضور ---
        function renderAttendance() {
            const date = document.getElementById('attDate').value;
            if(!date) return alert('اختر التاريخ');
            
            document.getElementById('attList').style.display = 'block';
            const tbody = document.querySelector('#attTable tbody');
            tbody.innerHTML = '';
            
            const savedAtt = appData.attendance[date] || [];
            
            appData.students.forEach(s => {
                tbody.innerHTML += `<tr>
                    <td>${s.name}</td>
                    <td><input type="checkbox" class="att-chk" data-id="${s.id}" ${savedAtt.includes(s.id) ? 'checked' : ''}></td>
                </tr>`;
            });
        }

        function saveAttendance() {
            const date = document.getElementById('attDate').value;
            const chks = document.querySelectorAll('.att-chk:checked');
            const presentIds = Array.from(chks).map(c => c.dataset.id);
            
            appData.attendance[date] = presentIds;
            saveData();
            showToast('تم حفظ سجل الحضور');
        }

        // --- الإعدادات ---
        function loadSettingsToUI() {
            document.getElementById('maxHw').value = appData.settings.hw;
            document.getElementById('maxPart').value = appData.settings.part;
            document.getElementById('maxMid').value = appData.settings.mid;
            document.getElementById('maxFinal').value = appData.settings.final;
        }

        function saveSettings() {
            const hw = parseFloat(document.getElementById('maxHw').value);
            const part = parseFloat(document.getElementById('maxPart').value);
            const mid = parseFloat(document.getElementById('maxMid').value);
            const final = parseFloat(document.getElementById('maxFinal').value);

            if(hw + part + mid + final !== 100) {
                return alert('يجب أن يكون مجموع الدرجات 100 بالضبط');
            }

            appData.settings = { hw, part, mid, final };
            saveData();
            showToast('تم حفظ إعدادات التوزيع');
        }

        // --- الإحصائيات ---
        function updateDashboard() {
            const totalStd = appData.students.length;
            document.getElementById('statCount').innerText = totalStd;
            
            if(totalStd === 0) return;

            let totalGrades = 0;
            let passCount = 0;

            appData.students.forEach(s => {
                const g = appData.grades[s.id] || {h:0, m:0, f:0};
                const sum = parseFloat(g.h||0) + parseFloat(s.partScore||0) + parseFloat(g.m||0) + parseFloat(g.f||0);
                totalGrades += sum;
                if(sum >= 50) passCount++;
            });

            document.getElementById('statPass').innerText = Math.round((passCount/totalStd)*100) + '%';
            document.getElementById('statAvg').innerText = (totalGrades/totalStd).toFixed(1);
        }

        // --- تصدير البيانات ---
        function exportExcel() {
            let csv = "\uFEFFالرقم الجامعي,الاسم,الواجبات,التفاعل,النصفي,النهائي,المجموع,الحالة\n";
            appData.students.forEach(s => {
                const g = appData.grades[s.id] || {};
                const total = parseFloat(g.h||0) + parseFloat(s.partScore||0) + parseFloat(g.m||0) + parseFloat(g.f||0);
                csv += `${s.id},${s.name},${g.h||0},${s.partScore||0},${g.m||0},${g.f||0},${total},${total>=50?'ناجح':'راسب'}\n`;
            });

            const blob = new Blob([csv], { type: 'text/csv;charset=utf-8;' });
            const url = window.URL.createObjectURL(blob);
            const a = document.createElement('a');
            a.setAttribute('hidden', '');
            a.setAttribute('href', url);
            a.setAttribute('download', `كشف_درجات_${new Date().toLocaleDateString()}.csv`);
            document.body.appendChild(a);
            a.click();
            document.body.removeChild(a);
        }

        function saveData() {
